CMAKE_MINIMUM_REQUIRED( VERSION 2.6 )
PROJECT( DateTime )
SET( DATETIME_ROOT_PATH ${PROJECT_BINARY_DIR}/.. )
SET( DATETIME_SOURCE_PATH ${DATETIME_ROOT_PATH}/src )
SET( DATETIME_BIN_PATH ${DATETIME_ROOT_PATH}/bin )

SET( EXECUTABLE_OUTPUT_PATH ${DATETIME_BIN_PATH} )

FILE(GLOB_RECURSE DATETIME_SOURCE_FILES ${DATETIME_SOURCE_PATH}/*.cpp)

SET( CMAKE_CXX_FLAGS "-std=c++0x -g -Wall" )

ADD_EXECUTABLE( DateTime ${DATETIME_SOURCE_FILES} )
